31 #define PX_PHYSICS_VERSION_NUMBER_H
32 
33 /*
34 VersionNumbers: The combination of these
35 numbers uniquely identifies the API, and should
36 be incremented when the SDK API changes. This may
37 include changes to file formats.
38 
39 This header is included in the main SDK header files
40 so that the entire SDK and everything that builds on it
41 is completely rebuilt when this file changes. Thus,
42 this file is not to include a frequently changing
43 build number. See BuildNumber.h for that.
44 
45 Each of these three values should stay below 255 because
46 sometimes they are stored in a byte.
47 */
52 #define PX_PHYSICS_VERSION_MAJOR 4
53 #define PX_PHYSICS_VERSION_MINOR 1
54 #define PX_PHYSICS_VERSION_BUGFIX 1
55 
60 #define PX_PHYSICS_VERSION ((PX_PHYSICS_VERSION_MAJOR<<24) + (PX_PHYSICS_VERSION_MINOR<<16) + (PX_PHYSICS_VERSION_BUGFIX<<8) + 0)

63 #endif // PX_PHYSICS_VERSION_NUMBER_H
